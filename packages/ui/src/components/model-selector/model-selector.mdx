# Model Selector

Dropdown selector for choosing AI models.

<Primary />

## Installation

```bash
pnpm dlx shadcn@latest add https://ui.vllnt.com/r/model-selector.json
```

## Import

```tsx
import { ModelSelector, ModelSelectorFilters } from '@vllnt/ui'
```

## Usage

```tsx
const models = [
  {
    description: "A description",
    id: "1",
    name: "value",
    pricing: "value",
    input: "value",
    output: "value",
  },
]

<ModelSelector
  models={models}
  onOpenChange={(open) => console.log(open)}
  onSelectModel={(modelId) => console.log(modelId)}
  selectedModelId="unique-id"
/>
```

<Canvas of={Stories.Default} sourceState="shown" />
